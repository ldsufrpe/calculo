\begin{align*} \integral{\sin^2{x}\cos^3{x}}{x}\amp = \integral{\sin^2{x}\cos^2{x}\cos{x}}{x}\\ \amp = \integral{\sin^{2}{x}(1-\sin^2{x})\cos{x}}{x} \amp \ctm{\cos^2{x}=1-\sin{x}^2}.\\ \amp =\integral{u^2(1-u^2)}{u}\amp \ctm{u=\sin{u},\, \dd u=\cos{x}\,\dd x}.\\ \amp= \integral{u^2-u^4}{u}\amp \ct{Integre na variável u} \\ \amp =\frac{1}{3}u^3-\frac{1}{5}u^5+C \amp \ct{Retorne a vaariável x}.\\ \amp= \frac{1}{3}\sin^3{x}-\frac{1}{5}\sin^5{x}+C \end{align*}
