\begin{equation*} \begin{array}{rcl} \displaystyle \int_{-7}^{2} \left( (0.5 x^2 + 7) - (x) \right) \, dx \amp =\amp \left( \dfrac{0.5 x^3}{3} + 7 x - \dfrac{x^2}{2} \right) \Big|_{-7}^{2} \\ \amp =\amp 144 \end{array} \end{equation*}
