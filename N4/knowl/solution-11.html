\begin{equation*} \frac{x^2}{x^2 + 4} = \frac{x^2+4-4}{x^2+4} = \frac{x^2 + 4}{x^2 + 4} - \frac{4}{x^2 + 4} = 1 - \frac{4}{x^2+4} \end{equation*}
