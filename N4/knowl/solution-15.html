\begin{equation*} \frac{x^2}{x^2 + 36} = \frac{x^2+36-36}{x^2+36} = \frac{x^2 + 36}{x^2 + 36} - \frac{36}{x^2 + 36} = 1 - \frac{36}{x^2+36} \end{equation*}
